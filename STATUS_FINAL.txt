🔨 COMANDOS ÚTEIS
═══════════════════════════════════════════════════════════════════════

Setup:
  chmod +x init.sh && ./init.sh         # Setup automático
  docker-compose up -d                  # Iniciar containers
  docker-compose down                   # Parar containers
  docker-compose down -v                # Resetar tudo

Backend:
  docker-compose exec backend bash      # Shell no backend
  docker-compose exec backend php artisan migrate
  docker-compose exec backend php artisan db:seed
  docker-compose exec backend php artisan tinker

Testes:
  docker-compose exec backend php artisan test
  docker-compose exec backend php artisan test --coverage

Logs:
  docker-compose logs -f                # Ver todos os logs
  docker-compose logs -f backend        # Ver logs backend
  docker-compose logs -f frontend       # Ver logs frontend
  docker-compose logs -f postgres       # Ver logs banco
